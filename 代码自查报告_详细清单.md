**修复建议**:
添加防抖工具函数：

```javascript
// utils/util.js
function debounce(fn, delay = 500) {
  let timer = null;
  return function(...args) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// 使用
submitForm: debounce(function() {
  // 提交逻辑
}, 1000)
```
